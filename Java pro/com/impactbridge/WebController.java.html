package com.impactbridge;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.ModelAttribute;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.PostMapping;
import java.util.List;

@Controller
public class WebController {

    @Autowired
    private VolunteerRepository volunteerRepository;

    // 1. Show the Registration Form
    @GetMapping("/register")
    public String showRegisterPage() {
        return "Register";
    }

    // 2. Handle the Form Submission (Saves Name, Email, Location, Skills, and Contact)
    @PostMapping("/register")
    public String registerVolunteer(@ModelAttribute Volunteer volunteer) {
        volunteerRepository.save(volunteer);
        return "redirect:/admin"; // Goes to dashboard after saving
    }

    // 3. Show the Admin Dashboard (Table View)
    @GetMapping("/admin")
    public String adminDashboard(Model model) {
        List<Volunteer> volunteers = volunteerRepository.findAll();
        model.addAttribute("volunteers", volunteers);
        return "admin";
    }

    // 4. Delete a Volunteer by ID
    @GetMapping("/delete/{id}")
    public String deleteVolunteer(@PathVariable Long id) {
        volunteerRepository.deleteById(id);
        return "redirect:/admin";
    }
}
